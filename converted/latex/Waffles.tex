\section{Waffles}

10 min. prep / 20 min. cooking

\begin{itemize}
\item 2 cups flour
\item 4 tsp baking powder
\item 1/2 tsp salt
\item 1/4 cup sugar
\item 2 eggs
\item 1/2 cup vegetable oil
\item 2 cups milk
\item 1 tsp vanilla
\end{itemize}

\subsection{Preparation}

\begin{enumerate}
\item Mix dry ingredients together
\item Add the rest of the ingredients and mix until smooth
\end{enumerate}
